Lifetimes

Lifetimes of a reference is about:

  • What regions of memory a reference could points to ?
  • When the regions are valid?

What regions of memory ?

fn main() {
    let r;
    {
        let x = 5;
        // before
        r = &x;
    }
    // after
    println!("r: {r}");     // compile error here
}

fn foo<'a>(x: &'a u32, y: &'a u32) -> &'a u32 {
    if (x > y) {
        x
    }
    else {
        y
    }
}

fn main() {
    let x = 12;
    let z: &u32 = {
        let y = 42;
        foo(&x, &y)
